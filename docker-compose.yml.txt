version: "3"

services:
  appserver:
    build: .
    container_name: appserver
    hostname: appserver
    restart: unless-stopped
    # command: ["npm", "run", "start"] # command: ["./wait-for-postgres.sh", "db", "npm", "run", "start"]
    depends_on:
      - db
    ports:
      - 8000:8000
    networks:
      - net
  
  nginxserver:
    build: nginx/
    container_name: nginxserver
    hostname: nginxserver
    restart: always
    ports:
      - 80:80
    depends_on:
      - appserver
      - db
    networks:
      - net
  
  db:
    image: postgres
    restart: unless-stopped
    container_name: dbpostgres
    hostname: db
    environment:
      POSTGRES_USER: dbpostgres
      POSTGRES_PASSWORD: dbpostgres
    volumes:
      - ./db:/var/lib/postgresql/data
    ports:
      - 5432:5432
    networks:
      - net

networks:
  net:
    driver: bridge
